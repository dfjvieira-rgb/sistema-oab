        if (sidebar.classList.contains('collapsed')) {

    function fazerBackup() { alert("Backup Exportado (Mock)"); }
    function importarBackup(input) { alert("Backup Importado (Mock)"); }
    function salvarAlteracoes() { alert("Altera√ß√µes Salvas (Mock)"); }
    function cancelarEdicao() { alert("Edi√ß√£o Cancelada (Mock)"); }
    function exportarTXT() { alert("Exportando para TXT (Mock)"); }
    function expandir(id) { alert(`Expandindo PDF ${id} (Mock)`); }
    
    window.onload = () => {
       renderAnnotations('caderno'); 
    }

    // ------------------------------------------------------------------
    // NOVO JS: Funcionalidade de Anota√ß√µes (Multi-tipo)
    // ------------------------------------------------------------------

    const PDF_PANEL = document.getElementById('pdf-panel');
    const ANNOTATIONS_MANAGER_PANEL = document.getElementById('annotations-manager-panel');
    const ANNOTATION_INPUT = document.getElementById('annotation-input');
    const ANNOTATIONS_LIST = document.getElementById('annotations-list');
    const CURRENT_TYPE_LABEL = document.getElementById('current-type-label');
    
    // Vari√°vel de controle para o tipo de anota√ß√£o atual
    let currentAnnotationType = 'caderno'; 

    /**
     * Abre o painel de gerenciamento de anota√ß√µes e alterna para uma aba espec√≠fica.
     * @param {string} type O tipo de anota√ß√£o a ser visualizado ('caderno', 'postit', 'dicas').
     */
    function openAnnotationsPanel(type = 'caderno') {
        PDF_PANEL.style.display = 'none';
        ANNOTATIONS_MANAGER_PANEL.style.display = 'flex';
        switchTab(type);
    }

    /**
     * Fecha o painel de anota√ß√µes e volta para os PDFs.
     */
    function closeAnnotationsPanel() {
        PDF_PANEL.style.display = 'flex';
        ANNOTATIONS_MANAGER_PANEL.style.display = 'none';
    }

    /**
     * Alterna entre as abas (tipos de anota√ß√£o).
     * @param {string} type O tipo de anota√ß√£o.
     */
    function switchTab(type) {
        currentAnnotationType = type;

        // Atualiza a visualiza√ß√£o da aba ativa
        document.querySelectorAll('.tab-button').forEach(btn => {
            btn.classList.remove('active');
        });
        event.target.classList.add('active'); // O bot√£o clicado fica ativo

        CURRENT_TYPE_LABEL.textContent = type.charAt(0).toUpperCase() + type.slice(1);
        renderAnnotations();
    }

    /**
     * Salva uma nova anota√ß√£o no LocalStorage com base no tipo atual.
     */
    function saveAnnotation() {
        const text = ANNOTATION_INPUT.value.trim();
        if (text === '') return;

        const STORAGE_KEY = `sistemaJoaquimAnnotations_${currentAnnotationType}`;
        let annotations = JSON.parse(localStorage.getItem(STORAGE_KEY)) || [];
        
        const newAnnotation = {
            id: Date.now(),
            text: text,
            date: new Date().toLocaleString('pt-BR')
        };

        annotations.push(newAnnotation);
        localStorage.setItem(STORAGE_KEY, JSON.stringify(annotations));
        
        ANNOTATION_INPUT.value = '';
        renderAnnotations();
    }

    /**
     * Renderiza todas as anota√ß√µes salvas na interface para o tipo atual.
     */
    function renderAnnotations() {
        ANNOTATIONS_LIST.innerHTML = '';
        const STORAGE_KEY = `sistemaJoaquimAnnotations_${currentAnnotationType}`;
        const annotations = JSON.parse(localStorage.getItem(STORAGE_KEY)) || [];

        annotations.forEach(annotation => {
            const item = document.createElement('div');
            // Aplica a classe de estilo visual baseada no tipo
            item.className = `annotation-list-item type-${currentAnnotationType}`; 
            item.innerHTML = `
                <p>${annotation.text}</p>
                <small>Criado em: ${annotation.date}</small>
                <button class="delete-annotation" onclick="deleteAnnotation(${annotation.id})">‚úñÔ∏è</button>
            `;
            ANNOTATIONS_LIST.appendChild(item);
        });
    }

    /**
     * Remove uma anota√ß√£o espec√≠fica pelo ID e tipo.
     * @param {number} id O ID (timestamp) da anota√ß√£o a ser removida.
     */
    function deleteAnnotation(id) {
        const STORAGE_KEY = `sistemaJoaquimAnnotations_${currentAnnotationType}`;
        let annotations = JSON.parse(localStorage.getItem(STORAGE_KEY)) || [];
        annotations = annotations.filter(annotation => annotation.id !== id);
        localStorage.setItem(STORAGE_KEY, JSON.stringify(annotations));
        renderAnnotations(); // Re-renderiza a lista ap√≥s a exclus√£o
    }
